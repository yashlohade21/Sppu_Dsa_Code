#include<iostream>
using namespace std;

struct node
{
   string data;
   struct node *next;
};
 class HashTable
 {
struct node *tbl[10],*nw,*t;

public:
HashTable()
{
    for(int i=0; i<10; i++)
         tbl[i]=NULL; 
}
void insertValue()
{
  int y,z;
  string x;
  cout<<"\n Enter The string :";
  cin>>x;
  nw=new (struct node);
  nw->next=NULL;
  nw->data=x;
  y=x[0]%10;
  if(tbl[y]==NULL)
     tbl[y]=nw;
    else
    {
	    t=tbl[y];
      while(t != NULL)
      {  
         if(t->next==NULL)
         {
            t->next=nw;
            break;
         }
         else
               t=t->next;
       }
     }
}

void display()
{
  cout<<"\n Content Of Hash Table Is:";
  for(int i=0; i<10; i++)
  {
       cout<<i<<" -> ";
       t=tbl[i];
       while(t!=NULL)
       {
          cout<<t->data<<" -> ";
          t=t->next;
       }
       cout<<"NULL"<<endl;
   }
}
void search()
{   int flag=0;

  t=tbl[y];
       while(t!=NULL )
       {
           if(t->data ==k)

          else
          t=t->next;
       }
       if(flag==0)
       cout<<endl<<"String is not found";
   }

};
int main(){
